类型断言, 由于接口是一般类型, 不知道具体类型, 如果要转换成具体类型,就需要类型断言.

![](https://ws3.sinaimg.cn/large/006tNc79ly1g21wdg6w5gj30du0bggm9.jpg)





给`Phone`增加一个`Call`方法

```go
type Usb interface {
	start()
	stop()
}
type Computer struct {
}


func (c Computer) work(usb Usb) {
	usb.start()
  // 重点: 断言检测
	if phone, ok := usb.(Phone); ok {
		phone.Call()
	}
	usb.stop()
}


// Phone 实现接口
type Phone struct {
}

func (p Phone) start() {
	fmt.Println("开始工作")
}
func (p Phone) stop() {
	fmt.Println("结束工作")
}
// 自己的Call方法
func (p Phone) Call() {
	fmt.Println("正在打电话")
}

func main() {
	phone := Phone{}
	computer := Computer{}
	computer.work(phone)
}

```
