```
function BinarySearch2()
{
	left = 0;
	right = count(array) -1;
	while(left <= right) {
		mid = (left + right)/2;
		if(array[mid] == target) {
			return mid;
		}elseif (array[mid] < target) {
			left = mid +1;
		}else{
			right = mid -1;
		}
	}
}
```

```java
//二分查找，递归版本
int BinarySearch2(int a[], int value, int low, int high)
{
    int mid = low+(high-low)/2;
    if(a[mid]==value)
        return mid;
    if(a[mid]>value)
        return BinarySearch2(a, value, low, mid-1);
    if(a[mid]<value)
        return BinarySearch2(a, value, mid+1, high);
}
```
